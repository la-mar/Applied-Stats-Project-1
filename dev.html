    x = df2[df2.char == 'Symp'].hours,

    mode = 'lines+markers',
    name = 'Symp'
)

data = [trace0, trace1]

layout = {
    "xaxis": {
        "title" : "Hours",
    },
    "yaxis": {
        "title" : "Count",
        "range" : [0, 12]
    }
}

fig = go.Figure(data=data
                , layout=layout)
plotly.offline.plot(fig
                    , filename=pth+'line.html' 
                    , image  = 'png'
                    , image_filename='p1-line')

2. Using the time variable and symptom status variables, generate a repeated measures model using the Compound Symmetry structure for the correlation that exists within people. Provide a plot of the residual diagnostics and comment on the validity of the assumptions.


                        proc mixed data=data plots=all;
                        class gender char hours subject age;
                        model ifitm3 = age hours age*hours / solution;
                        repeated hours/ type=CS subject=subject;
                        lsmeans age*hours / pdiff tdiff adjust=Tukey;
                        estimate 'Treatment v. Control' hours 1 -1;
                        run;
The residuals appear to be normally distributed, but there are strong indications of a lack of independence, as seen in the residual scatter plot.
